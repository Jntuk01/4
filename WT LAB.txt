        // Convert strings to numbers and trim spaces
        $array1 = array_map('intval', array_map('trim', $array1));

        $mergedArray = array_merge($array1, $array2);

EXPERIMENT 5 - PHP SCRIPT TO PRINT PRIME NUMBERS 


<?php
function isPrime($num) {
    if ($num <= 1) {
        return false;
    }
    for ($i = 2; $i <= sqrt($num); $i++) {
        if ($num % $i == 0) {
            return false;
        }
    }
    return true;
}

function printPrimes($start, $end) {
    echo "Prime numbers between $start and $end are:<br>";
    for ($i = $start; $i <= $end; $i++) {
        if (isPrime($i)) {
            echo $i . " ";
        }
    }
    echo "<br>";
}
?>

<form method="post">
    Start: <input type="number" name="start" required>
    End: <input type="number" name="end" required>
    <input type="submit" value="Show Primes">
</form>

    transition: background-color 0.3s, border-radius 0.3s;
  }

<script>
  let shape = true;

  const bean = document.getElementById('bean');

  function getRandomColor() {
    return '#' + Math.floor(Math.random() * 16777215).toString(16);
  }

  function updateBean() {
    bean.style.backgroundColor = getRandomColor();
    bean.style.borderRadius = shape ? '0%' : '50%';
  }

  updateBean();

  bean.addEventListener('click', () => {
    shape = !shape;
    updateBean();
  });
</script>

</body>
</html>
